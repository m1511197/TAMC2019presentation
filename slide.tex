

\section{Oritatami system}
\subsection{What is oritatami system?}
\begin{frame}\frametitle{What is oritatami system?}
\begin{block}{}
Oritatami system is a mathematical model for cotranscriptional folding(CF). (Geary, Meunier, Schabanel and Seki.)
\end{block}

\begin{figure}[htbp]
  \begin{center}
    \leavevmode
    \includegraphics[width=0.4\textwidth]{RNA.png} 
    \caption{RNA tile is self-assembled (RNA Origami)}
  \end{center}
\end{figure}

(C. Geary, P.W.K. Rothemund and E.S. Andersen. Science 345(6198), 2014)

\end{frame}

\section{Turing universal}
\begin{frame}\frametitle{Turing universal}

\begin{block}{Theorem (C. Geary \textit{et \ al}. ISAAC, 2018)}
Oritatami system at $delay \ \delta = 3$ which employs 542 types of beads is Turing universal.
\end{block}

\begin{block}{Theorem}
Polynomial length of conformations $\longrightarrow$ Non-Turing-universal
\end{block}

\end{frame}







\section{Problem}
\begin{frame}\frametitle{Problem}

\begin{block}{Problem}
Give an upper bound on the length of a transcript of a $delay \ \delta$, $arity \ \alpha$ deterministic oritatami system by a function in $\delta$, $\alpha$, and seed \textit{n}.
\end{block}

\begin{block}{Oritatami System}
\ \ input : $delay \ \delta$, $arity \ \alpha$, seed, rule, transcript\\
output : conformation
\end{block}

\end{frame}







\section{Why unary?}
\begin{frame}\frametitle{Why unary?}

\begin{block}{}
Because we considered the unary oritatami system is good for a first step towards the characterization of non-Turing-universal oritatami systems.
\end{block}

\end{frame}
